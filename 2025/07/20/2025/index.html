空值合并操作符 ??
  • 语法：value1 ?? value2
  • 作用：如果 value1 是 null 或 undefined，则返回 value2；否则返回 value1。
  • 与逻辑或操作符||的区别：
    • || 返回第一个真值（如 0、''、false 会被视为假值）。
    • ?? 仅在值为 null 或 undefined 时才返回默认值，更适合设置默认配置。

#set_error_handler：处理 PHP 错误（非异常）
#设置错误处理函数。它能够捕获 PHP 程序运行过程中产生的警告（E_WARNING）、通知（E_NOTICE）等非致命错误。
set_error_handler(function ($errno, $errstr, $errfile, $errline) use (&$currentClass) {
    if (strpos($errstr, "Mzzz") !== false) {
        echo "[FOUND] Class: $currentClass\n";
        echo "  └── Message: $errstr\n";
    }
    return true;
});

foreach ($classes as $class) {
    $currentClass = $class;
    try {
        $refClass = new ReflectionClass($class);
        if ($refClass->isInstantiable()) {
            #isInstantiable()方法用于判断一个类是否能够被实例化，像抽象类、接口就不可以实例化。
            $instance = $refClass->newInstance("Mzzz");
            #newInstance("Mzzz")会尝试用"Mzzz"作为参数来创建类的实例。如果类的构造函数不接受参数或者需要多个参数，这一步就可能会失败。
            $s = serialize($instance);
            unserialize($s);
        }
        # catch (Throwable $e)：处理异常（Exception/Error）
    } catch (Throwable $e) {
        #catch (Throwable $e)可以捕获所有类型的异常和错误，包括 PHP 7 引入的 Error 类。
        if (strpos($e->getMessage(), "Mzzz") !== false) {
            echo "[FOUND] Class: $class\n";
