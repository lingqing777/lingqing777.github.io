import hashlib
import time


def generate_signature(username, password):
    # 1. 生成MD5哈希并截断
    md5_user = hashlib.md5(username.encode('utf-8')).hexdigest()[:6]
    md5_pass = hashlib.md5(password.encode('utf-8')).hexdigest()[:6]

    # 2. 获取时间戳（毫秒级）
    timestamp = str(int(time.time() * 1000))

    # 3. 拼接签名要素
    secret = 'easy_signin'
    sign_str = md5_user + md5_pass + timestamp + secret

    # 4. 生成最终签名
    return hashlib.md5(sign_str.encode('utf-8')).hexdigest()
